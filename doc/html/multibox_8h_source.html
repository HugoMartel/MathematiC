140template <typename Scalar>
141class IntBox : public MultiBox {
142public:
143 IntBox(Widget *parent, Scalar value = (Scalar) 0) : MultiBox(parent) {
144 setDefaultValue("0");
145 setFormat(std::is_signed<Scalar>::value ? "[-]?[0-9]*" : "[0-9]*");
146 setValueIncrement(1);
147 setMinMaxValues(std::numeric_limits<Scalar>::lowest(), std::numeric_limits<Scalar>::max());
148 setValue(value);
149 setSpinnable(false);
150 }
151
152 Scalar value() const {
153 std::istringstream iss(MultiBox::value());
154 Scalar value = 0;
155 iss >> value;
156 return value;
157 }
158
159 void setValue(Scalar value) {
160 Scalar clampedValue = std::min(std::max(value, mMinValue),mMaxValue);
161 MultiBox::setValue(std::to_string(clampedValue));
162 }
163
164 void setCallback(const std::function<void(Scalar)> &cb) {
165 MultiBox::setCallback(
166 [cb, this](const std::string &str) {
167 std::istringstream iss(str);
168 Scalar value = 0;
169 iss >> value;
170 setValue(value);
171 cb(value);
172 return true;
173 }
174 );
175 }
176
177 void setValueIncrement(Scalar incr) {
178 mValueIncrement = incr;
179 }
180 void setMinValue(Scalar minValue) {
181 mMinValue = minValue;
182 }
183 void setMaxValue(Scalar maxValue) {
184 mMaxValue = maxValue;
185 }
186 void setMinMaxValues(Scalar minValue, Scalar maxValue) {
187 setMinValue(minValue);
188 setMaxValue(maxValue);
189 }
190
191 virtual bool mouseButtonEvent(const Vector2i &p, int button, bool down, int modifiers) override {
192 if ((mEditable || mSpinnable) && down)
193 mMouseDownValue = value();
194
195 SpinArea area = spinArea(p);
196 if (mSpinnable && area != SpinArea::None && down && !focused()) {
197 if (area == SpinArea::Top) {
198 setValue(value() + mValueIncrement);
199 if (mCallback)
200 mCallback(mValue);
201 } else if (area == SpinArea::Bottom) {
202 setValue(value() - mValueIncrement);
203 if (mCallback)
204 mCallback(mValue);
205 }
206 return true;
207 }
208
209 return MultiBox::mouseButtonEvent(p, button, down, modifiers);
210 }
211 virtual bool mouseDragEvent(const Vector2i &p, const Vector2i &rel, int button, int modifiers) override {
212 if (MultiBox::mouseDragEvent(p, rel, button, modifiers)) {
213 return true;
214 }
215 if (mSpinnable && !focused() && button == 2 /* 1 << GLFW_MOUSE_BUTTON_2 */ && mMouseDownPos.x != -1) {
216 int valueDelta = static_cast<int>((p.x - mMouseDownPos.x) / float(10));
217 setValue(mMouseDownValue + valueDelta * mValueIncrement);
218 if (mCallback)
219 mCallback(mValue);
220 return true;
221 }
222 return false;
223 }
224 virtual bool scrollEvent(const Vector2i &p, const Vector2f &rel) override {
225 if (Widget::scrollEvent(p, rel)) {
226 return true;
227 }
228 if (mSpinnable && !focused()) {
229 int valueDelta = (rel.y > 0) ? 1 : -1;
230 setValue(value() + valueDelta*mValueIncrement);
231 if (mCallback)
232 mCallback(mValue);
233 return true;
234 }
235 return false;
236 }
237private:
238 Scalar mMouseDownValue;
239 Scalar mValueIncrement;
240 Scalar mMinValue, mMaxValue;
241};
